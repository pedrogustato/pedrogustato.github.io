def Inverter( img ):

    y2 = input('y2: ')
    x2 = input('x2: ')

    rectangle = img[int(y1):int(y2), int(x1):int(x2)]
    altura, largura = rectangle.shape

    for i in range (0, altura):
        for j in range(0, largura):
            rectangle.itemset((i, j), 255-(rectangle.item(i, j)))
    img[int(y1):int(y2), int(x1):int(x2)] = rectangle

    return img 

img = cv.imread("negativo.jpg", 0)
img_neg = Inverter(img)
showImage(img_neg)

O código é dividido em 3 etapas onde a primeira é representada pela função showImage() onde a imagem resultante é plotada e escrita na memória e a segunda e terceira estão embutidas na função Inverter() que consistem respectivamente em saber os pontos de começo e final, da área que será invertida e da etapa onde se é invertido de fato.

from tempfile import tempdir

    cv.imwrite("cortado.png", img)

def trocaRegioes( img ):
    cv.imwrite("troca.png", img)
    tempImg = cv.imread("troca.png", 0)

    altura, largura = img.shape

    tempImg[0:int(altura/2), 0:int(largura/2)] = img[int(altura/2):altura, int(largura/2):largura]
    tempImg[0:int(altura/2), int(largura/2):largura] = img[int(altura/2):altura, 0:int(largura/2)]
    tempImg[int(altura/2):altura, 0:int(largura/2)] = img[0:int(altura/2), int(largura/2):largura]
    tempImg[int(altura/2):altura, int(largura/2):largura] = img[0:int(altura/2), 0:int(largura/2)]

    return tempImg


img = cv.imread("negativo.jpg", 0)
showImage(trocaRegioes(img))
